#version 330 core

layout(location = 0) in vec3 position;
layout(location = 1) in vec3 aNormal;
layout(location = 2) in vec2 aTexCoord;

out vec2 TexCoord;
out vec3 vNormal;
out vec3 FragPos;

uniform mat4 model;
uniform mat4 view;
uniform mat4 proj;

void main()
{
	FragPos = vec3(view * model * vec4(position, 1.0));
    vNormal = normalize(vec3(view * model * vec4(aNormal, 0.0)));  
    TexCoord = aTexCoord;
    
    gl_Position = proj * view * model * vec4(position, 1.0);
}
